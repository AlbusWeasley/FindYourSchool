async function register() { const username = document.getElementById("username").value; const password = document.getElementById("password").value; const response = await fetch("/api/auth/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username, password }) }); const data = await response.json(); console.log("注册响应:", data); if (response.ok) { alert("注册成功，请登录！"); } else { alert("注册失败：" + data.error); } }
